\begin{thm}[H\"older's inequality]
    Let $p,q\in[1, \infty]$ satisfy
    $\frac{1}{p}+\frac{1}{q}=1$
    and let $f$ and $g$ be measurable functions.
    Then
    \[
    \|fg\|_{1}
    \le
    \|f\|_{p}\|g\|_{q}
    \]
\end{thm}

\begin{proof}
    The case $p=\infty$ is trivial.
    Assume $p,q<\infty$.
    We first prove the following auxiliary inequality:
    If $s,t>0$, then
    \[
    st\le\frac{s^p}{p}+\frac{t^q}{q}.
    \]
    Let $\sigma,\tau\in\mathbb{R}$, and set $s=e^{\sigma},t=e^{\tau}$.
    Since $e^x$ is convex and
    \[
    \frac{1}{p}+\frac{1}{q}=1
    \]
    we obtain
    \[
    e^{\sigma+\tau} \le \frac{e^{p\sigma}}{p}+\frac{e^{q\tau}}{q}.
    \]
    Hence
    \[
    st\le\frac{s^p}{p}+\frac{t^q}{q}.
    \]
    Using this inequality for $(s,t)=(|f|,|g|)$, we obtain
    \[
    |fg|\le\frac{|f|^p}{p}+\frac{|g|^q}{q}.
    \]
    Integrating both sides, we obtain
    \begin{equation}
    \|fg\|_{1}\le\frac{\|f\|_{p}^{p}}{p}+\frac{\|g\|_{q}^{q}}{q}.
    \label{eq:holder-key}
    \end{equation}
    Setting $C=\|f\|_{p}\|g\|_{q},\tilde{f}=\frac{f}{\|f\|_{p}},\tilde{g}=\frac{g}{\|g\|_{q}}$,
    we obtain
    \[
    \|\tilde{f}\|_{p}^{p}=\|\tilde{g}\|_{q}^{q}=1
    \]
    The case $C=\infty$ is trivial.
    So, let $C<\infty$.
    Then, 
    \[
    C^{-1}\|fg\|_{1}=\|\tilde{f}\tilde{g}\|_{1}\overset{\eqref{eq:holder-key}}\le\frac{\|\tilde{f}\|_{p}^{p}}{p}+\frac{\|\tilde{g}\|_{q}^{q}}{q}=\frac1p+\frac1q=1.
    \]
    Multiplying both sides by $C$, we obtain
    \[
    \|fg\|_1 \le \|f\|_p\|g\|_q.
    \]


\end{proof}

\begin{thm}[Minkowski's inequality]
    Let $p\in[1,\infty]$, and $f,g:S\to\mathbb{C}$ is measurable function. Then,
    \[
    \|f+g\|_{p}\le\|f\|_{p}+\|g\|_{p}
    \]
\end{thm}

\begin{proof}
    we first prove the following auxiliary inequalities:
    Set $h=|f+g|$. Then,
    \begin{equation}
         \|h\|_{p}^{p}\le2^{p-1}(\|f\|_{p}^{p}+\|g\|_{p}^{p}).
        \label{eq:Minkowski-key1}
    \end{equation}
    \begin{equation}
        \|h\|_{p}^{p}\le(\|f\|_{p}+\|g\|_{p})\|h\|_{p}^{p-1}.
        \label{eq:Minkowski-key2}
    \end{equation}
    Proof of \eqref{eq:Minkowski-key1}:
    Note that following inequality:
    \[
    s,t\in\mathbb{C}, |s+t|^p\le2^{p-1}(|s|^{p}+|t|^{p})
    \]
    By triangle inequality, 
    \[
    |s+t|\le|s|+|t|
    \]
    So, it suffices to prove the claim in the case $s,t\in[0,\infty)$.
    Since $x\mapsto{x^{p}}$ is convex, we obtain
    \[
    \left(\frac{s+t}{2}\right)^p\le\frac{s^p}{2}+\frac{t^p}{2}
    \]
    Multiplying both sides by $2^p$, we obtain
    \[
    (s+t)^p\le2^{p-1}(s^p+t^p)
    \]
    Applying the above inequality pointwise to $s=f(x),t=g(x)$, and integrate both sides, we obtain (2).\\
    Proof of \eqref{eq:Minkowski-key2}:
    Note that 
    \[
    \|h\|_{p}^{p}=\|h^{p}\|_{1}=\|hh^{p-1}\|_{1}\le\|fh^{p-1}\|_{1}+\|gh^{p-1}\|_{1}
    \],
    \[
    \|h^{p-1}\|_{q}=\|h\|_{p}^{p-1}(\text{Set}q=\frac{p}{p-1})
    \],
    we obtain
    \[
    \|fh^{p-1}\|_{1}+\|gh^{p-1}_{1}\overset{H\"older}\le\|f\|_{p}\|h^{p-1}\|_{q}+\|g\|_{p}\|h^{p-1}\|_{q}=(\|f\|_{p}+\|g\|_{p})\|h\|_{p}^{p-1}
    \]
    We proved (2), (3). So, we prove the desired inequality, using (1), (2).
    The case $\|f\|_{p}+\|g\|_{p}=\infty$ is trivial.
    Let $\|f\|_{p}+\|g\|_{p}<\infty$, we obtain $\|h\|_{p}<\infty$ by (2).Then, dividing both sides of (3) by $\|h\|_{p}^{p-1}$, we obtain
    \[
    \|f+g\|_{p}\le\|f\|_{p}+\|g\|_{p}.
    \]



\end{proof}
